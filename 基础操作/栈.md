## 栈

[P5788 【模板】单调栈 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)](https://www.luogu.com.cn/problem/P5788)

```c++
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N=3e6+10;
int f[N],a[N];
int n;
stack<int> st;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(int i=n;i;i--)
	{
		while(!st.empty() && a[st.top()]<=a[i]) st.pop();
		f[i]=st.empty()? 0: st.top();
		st.push(i);
	}
	for(int i=1;i<=n;i++)
	{
		printf("%d ",f[i]);
	}
	printf("\n");
	return 0;
}
```
