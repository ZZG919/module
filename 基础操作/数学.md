## 最大公因数和最小公倍数

```c++
#include <iostream>
#include <vector>
using namespace std; 
//辗转相除法
int gcd(int a,int b)
{
	return a%b==0? b : gcd(b,a%b);	
} 
//最小公倍数:乘积/最小公倍数
int lcm(int a,int b)
{
	return a*b/gcd(a,b); 
} 
int main()
{
	int a=69,b=9;
	cout << gcd(a,b) << endl;
	cout << lcm(a,b) << endl;
	return 0;
}
```
