# 最短路径

## Floyd

题库：[P3371 【模板】单源最短路径（弱化版）使用floyd能过7题](https://www.luogu.com.cn/problem/P3371)

```c++
#include<bits/stdc++.h>
using namespace std;
const int inf=INT_MAX; 
const int N= 10005;
int a[N][N],n,m,s;
inline void floyd()
{
    for(int k=1;k<=n;k++)
	{
        for(int i=1;i<=n;i++)
		{
            if(i==k||a[i][k]==inf) continue;
			for(int j=1;j<=n;j++)
			{
				if(k==j || a[k][j]==inf) continue;
                a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
            }
        }
    }
}
int main()
{
	scanf("%d%d%d",&n,&m,&s);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)
            a[i][j]=inf;
    for(int i=1,u,v,w;i<=m;i++)
	{
        scanf("%d%d%d",&u,&v,&w);
        a[u][v]=min(a[u][v],w);
    }
    floyd();
    a[s][s]=0;
    for(int i=1;i<=n;i++)
    {
        printf("%d ",a[s][i]);
    }
    return 0;
}
```
