# 最短路径

## Floyd

题库：[P3371 【模板】单源最短路径（弱化版） - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)](https://www.luogu.com.cn/problem/P3371)

```c++
#include <iostream>
#include <vector>
#include <string> 
#include <bits/stdc++.h> 
using namespace std;
const int N=10001;
int g[N][N];
int n,m,start; //点个数,边个数，出发点 
void Floyd()
{
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			// 借助k这一跳然后更新i和j的距离 
			for(int k=1;k<=n;k++)
			{
				if(i!=j && i!=k && j!=k && 
					g[i][k]!=-1 && g[k][j]!=-1 
					&& g[i][k]+g[k][j]<g[i][j])
				{
					g[i][j]=g[i][k]+g[k][j];
				}
			}
		}
	 } 
}
int main()
{
	ios::sync_with_stdio(false);
	cin >> n >> m >> start;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		//到达不了设置为-1 
			g[i][j]=-1;	 
		//与自身的距离为0 
		g[i][i]=0;
	}
	//默认输入的节点为[1,n]中的所有整数取值 
	for(int i=0;i<m;i++)
	{
		int u,v,e; 
		cin >> u >> v>> e;
		g[u][v]=e;
	}
	Floyd();
	for(int i=1;i<=n;i++)
	{
		cout << g[start][i] << endl;
	 } 
	return 0 ;
 } 
```
