# 并查集

## 基础并查集

### 题库：

[冗余连接]([684. 冗余连接 - 力扣（LeetCode） (leetcode-cn.com)](https://leetcode-cn.com/problems/redundant-connection/))

```c++
class Solution {
public:
    vector<int> id;
    vector<int> ans;
    int find(int p)
    {
        while(p!=id[p])
        {
            p=id[p];
        }
        return p;
    }
    inline void connect(int p,int q)
    {
        id[find(p)]= find(q);
    }
    int isconnect(int p,int q)
    { 
        return find(p)==find(q);
    } 
    vector<int> findRedundantConnection(vector<vector<int>>& edges) {
        id.resize(1001);
        iota(id.begin(),id.end(),0);
        for(auto &e:edges)
        {
            int u=e[0],v=e[1];
            if(isconnect(u,v))
            {
                ans=e;
            }
            else connect(u,v);
        }
        return ans;
    }
};
```

## 并查集优化

### 题库：

[冗余连接]([684. 冗余连接 - 力扣（LeetCode） (leetcode-cn.com)](https://leetcode-cn.com/problems/redundant-connection/))

```c++
class Solution {
public:
    vector<int> id,size;
    vector<int> ans;
    int find(int p)
    {
        while(p!=id[p])
        {
            //路径压缩
            id[p]=id[id[p]];
            p=id[p];
        }
        return p;
    }
    inline void connect(int p,int q)
    {
        int i=find(p),j=find(q);
        if(i==j) return ;
        //将深度小的集合放在深度大的集合的下一跳
        if(size[i]<size[j])
        {
            id[i]=j;
            size[j]+=size[i];
        }
        else{
            id[j]=i;
            size[i]+=size[j];
        }
    }
    int isconnect(int p,int q)
    { 
        return find(p)==find(q);
    } 
    vector<int> findRedundantConnection(vector<vector<int>>& edges) {
        id.resize(1001);
        size.resize(1001,1);
        iota(id.begin(),id.end(),0);
        for(auto &e:edges)
        {
            int u=e[0],v=e[1];
            if(isconnect(u,v))
            {
                ans=e;
            }
            else connect(u,v);
        }
        return ans;
    }
};
```
